     6вҶ’## 1. Database Schema (PostgreSQL)
     7вҶ’
     8вҶ’### Core Principle
     9вҶ’One database, fully normalized, with row-level security (RLS) policies. The CRM is not a separate system вҖ” it IS the membership database. Every user interaction, token lifecycle event, email send, payment, and referral is a row in PostgreSQL.
    10вҶ’
    11вҶ’### Schema Overview
    12вҶ’```
    13вҶ’в”ңв”Җв”Җ users                  # All accounts (free, paid, token holders)
    14вҶ’в”ңв”Җв”Җ tiers                  # Membership tier definitions
    15вҶ’в”ңв”Җв”Җ subscriptions          # Stripe-linked paid subscriptions
    16вҶ’в”ңв”Җв”Җ tokens                 # Influencer access tokens
    17вҶ’в”ңв”Җв”Җ token_invitations      # Who was invited, when, status
    18вҶ’в”ңв”Җв”Җ referrals              # Referral tracking (who referred whom)
    19вҶ’в”ңв”Җв”Җ content                # All content (reports, opinions, studies, videos)
    20вҶ’в”ңв”Җв”Җ content_access_rules   # Which tier can access what content type
    21вҶ’в”ңв”Җв”Җ content_views          # Audit log of every content view
    22вҶ’в”ңв”Җв”Җ emails                 # All emails sent (newsletter + transactional)
    23вҶ’в”ңв”Җв”Җ email_subscriptions    # User newsletter preferences
    24вҶ’в”ңв”Җв”Җ alerts                 # Breaking news alerts
    25вҶ’в”ңв”Җв”Җ alert_deliveries       # Who received what alert, when
    26вҶ’в”ңв”Җв”Җ events                 # Virtual and in-person events
    27вҶ’в”ңв”Җв”Җ event_registrations    # Who signed up for what
    28вҶ’в”ңв”Җв”Җ analyst_calls          # Executive tier 1:1 calls
    29вҶ’в”ңв”Җв”Җ vavatar_videos         # AI-generated video metadata
    30вҶ’в”ңв”Җв”Җ crm_activities         # The CRM: every interaction logged
    31вҶ’в”ңв”Җв”Җ crm_tasks              # Follow-up tasks for team
    32вҶ’в”ңв”Җв”Җ crm_segments           # Dynamic user segments
    33вҶ’в””в”Җв”Җ audit_log              # Immutable change log
    34вҶ’```

    38вҶ’### Table: users
    39вҶ’```sql
    40вҶ’CREATE TABLE users (
    41вҶ’    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    42вҶ’    email           VARCHAR(255) UNIQUE NOT NULL,
    43вҶ’    password_hash   VARCHAR(255),           -- NULL for OAuth/token users
    44вҶ’    auth_method     VARCHAR(20) NOT NULL DEFAULT 'password', -- password | oauth_linkedin | token
    45вҶ’    
    46вҶ’    -- Profile
    47вҶ’    first_name      VARCHAR(100),
    48вҶ’    last_name       VARCHAR(100),
    49вҶ’    title           VARCHAR(100),           -- Job title
    50вҶ’    organization    VARCHAR(200),
    51вҶ’    country         VARCHAR(2),             -- ISO-3166
    52вҶ’    language_pref   VARCHAR(5) DEFAULT 'en', -- en | de | fr | it
    53вҶ’    timezone        VARCHAR(50) DEFAULT 'Europe/Zurich',
    54вҶ’    
    55вҶ’    -- Tier & Access
    56вҶ’    current_tier_id UUID REFERENCES tiers(id),
    57вҶ’    access_type     VARCHAR(20) NOT NULL DEFAULT 'free', -- free | paid | token | trial
    58вҶ’    token_id        UUID REFERENCES tokens(id),
    59вҶ’    
    60вҶ’    -- Engagement Scoring (CRM)
    61вҶ’    engagement_score INT DEFAULT 0,          -- 0-100, auto-calculated
    62вҶ’    last_active_at  TIMESTAMPTZ,
    63вҶ’    login_count     INT DEFAULT 0,
    64вҶ’    content_views_30d INT DEFAULT 0,
    65вҶ’    
    66вҶ’    -- Status
    67вҶ’    status          VARCHAR(20) DEFAULT 'active', -- active | paused | churned | banned
    68вҶ’    email_verified  BOOLEAN DEFAULT FALSE,
    69вҶ’    created_at      TIMESTAMPTZ DEFAULT NOW(),
    70вҶ’    updated_at      TIMESTAMPTZ DEFAULT NOW(),
    71вҶ’    
    72вҶ’    -- Stripe
    73вҶ’    stripe_customer_id VARCHAR(100),
    74вҶ’    stripe_subscription_id VARCHAR(100)
    75вҶ’);
    76вҶ’
    77вҶ’CREATE INDEX idx_users_tier ON users(current_tier_id);
    78вҶ’CREATE INDEX idx_users_token ON users(token_id);

   653вҶ’**Content Access Check** (runs on every content request):
   654вҶ’```typescript
   655вҶ’// lib/access.ts
   656вҶ’async function checkContentAccess(userId: string, contentSlug: string) {
   657вҶ’  const content = await db.query(
   658вҶ’    `SELECT c.*, t.early_access_hours 
   659вҶ’     FROM content c 
   660вҶ’     JOIN tiers t ON c.min_tier_id = t.id 
   661вҶ’     WHERE c.slug = $1`, [contentSlug]
   662вҶ’  );
   663вҶ’  
   664вҶ’  const user = await db.query(
   665вҶ’    `SELECT u.*, t.early_access_hours as user_early_access
   666вҶ’     FROM users u 
   667вҶ’     JOIN tiers t ON u.current_tier_id = t.id 
   668вҶ’     WHERE u.id = $1`, [userId]
   669вҶ’  );
   670вҶ’  
   671вҶ’  // Reverse timewall logic
   672вҶ’  if (!content.min_tier_id) return { access: 'full' }; // Free content
   673вҶ’  
   674вҶ’  const userTierLevel = getTierLevel(user.current_tier_id);
   675вҶ’  const contentTierLevel = getTierLevel(content.min_tier_id);
   676вҶ’  
   677вҶ’  if (userTierLevel > contentTierLevel) {
   678вҶ’    return { access: 'full', earlyAccess: true }; // Higher tier = full + early
   679вҶ’  }
   680вҶ’  
   681вҶ’  if (userTierLevel === contentTierLevel) {
   682вҶ’    // Check if content has aged into this tier
   683вҶ’    const hoursSincePublish = (now - content.published_at) / 3600000;
   684вҶ’    if (hoursSincePublish >= content.early_access_hours) {
   685вҶ’      return { access: 'full' };
   686вҶ’    }
   687вҶ’    return { access: 'preview', reason: 'early_access' };
   688вҶ’  }
   689вҶ’  
   690вҶ’  return { access: 'preview', reason: 'tier_upgrade_needed' };
   691вҶ’}

   694вҶ’**Engagement Scoring** (runs daily via cron):
   695вҶ’```typescript
   696вҶ’// lib/scoring.ts
   697вҶ’async function calculateEngagementScore(userId: string): Promise<number> {
   698вҶ’  const last30d = await db.query(`
   699вҶ’    SELECT 
   700вҶ’      COUNT(DISTINCT cv.content_id) * 5 as content_score,
   701вҶ’      COUNT(CASE WHEN cv.view_type = 'download' THEN 1 END) * 10 as download_score,
   702вҶ’      COUNT(CASE WHEN a.activity_type = 'email_open' THEN 1 END) * 1 as email_score,
   703вҶ’      COUNT(CASE WHEN a.activity_type = 'email_click' THEN 1 END) * 3 as click_score,
   704вҶ’      COUNT(CASE WHEN a.activity_type = 'login' THEN 1 END) * 2 as login_score,
   705вҶ’      COUNT(CASE WHEN a.activity_type = 'call_attended' THEN 1 END) * 25 as call_score,
   706вҶ’      MAX(a.occurred_at) as last_active
   707вҶ’    FROM users u
   708вҶ’    LEFT JOIN content_views cv ON cv.user_id = u.id AND cv.accessed_at > NOW() - INTERVAL '30 days'
   709вҶ’    LEFT JOIN crm_activities a ON a.user_id = u.id AND a.occurred_at > NOW() - INTERVAL '30 days'
   710вҶ’    WHERE u.id = $1
   711вҶ’    GROUP BY u.id
   712вҶ’  `, [userId]);
   713вҶ’  
   714вҶ’  const score = Math.min(100, (
   715вҶ’    (last30d.content_score || 0) +
   716вҶ’    (last30d.download_score || 0) +
   717вҶ’    (last30d.email_score || 0) +
   718вҶ’    (last30d.click_score || 0) +
   719вҶ’    (last30d.login_score || 0) +
   720вҶ’    (last30d.call_score || 0)
   721вҶ’  ));
   722вҶ’  
   723вҶ’  // Decay for inactivity: -10 points per week of no login
   724вҶ’  const daysSinceActive = Math.floor((now - last30d.last_active) / 86400000);
   725вҶ’  const decay = Math.floor(daysSinceActive / 7) * 10;
   726вҶ’  
   727вҶ’  await db.query(`UPDATE users SET engagement_score = $1 WHERE id = $2`,
   728вҶ’    [Math.max(0, score - decay), userId]
   729вҶ’  );
   730вҶ’  
   731вҶ’  return Math.max(0, score - decay);
   732вҶ’}

   798вҶ’

   803вҶ’}

   842вҶ’### Key CRM Query Examples
   843вҶ’
   844вҶ’```sql
   845вҶ’-- Segment: High-value churn risk (engagement < 20, paid tier, renewal in 30 days)
   846вҶ’SELECT u.*, s.current_period_end
   847вҶ’FROM users u
   848вҶ’JOIN subscriptions s ON s.user_id = u.id
   849вҶ’WHERE u.engagement_score < 20
   850вҶ’  AND u.access_type = 'paid'
   851вҶ’  AND s.current_period_end BETWEEN NOW() AND NOW() + INTERVAL '30 days'
   852вҶ’  AND s.status = 'active';
   853вҶ’
   854вҶ’-- Segment: Ambassador tokens with 0 referrals (nudge candidates)
   855вҶ’SELECT t.*, u.last_active_at
   856вҶ’FROM tokens t
   857вҶ’JOIN users u ON u.token_id = t.id
   858вҶ’WHERE t.token_type = 'ambassador'
   859вҶ’  AND t.status = 'active'
   860вҶ’  AND t.referrals_count = 0
   861вҶ’  AND t.activated_at < NOW() - INTERVAL '30 days';
   862вҶ’
   863вҶ’-- Revenue dashboard: MRR by tier
   864вҶ’SELECT t.name, COUNT(s.id) as subs, SUM(s.amount_chf) as mrr
   865вҶ’FROM subscriptions s
   866вҶ’JOIN tiers t ON t.id = s.tier_id
   867вҶ’WHERE s.status = 'active'
   868вҶ’GROUP BY t.name;
